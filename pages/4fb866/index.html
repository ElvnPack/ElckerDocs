环境配置

为适配当下开发场景的需求，EAdmin 将自身环境区分为3个环境，

  • 开发环境，提供mock支持，帮助开发人员在服务未完成前进行项目的开发，

  • 测试环境用于前端人员与后端服务的本地测试，

  • 生产环境用于线上。

各个环境的详细配置内容参见：

# 全局环境配置

# 项目的页面文件地址配置 参见 ：https://www.npmjs.com/package/vite-plugin-pages
VITE_PAGES_ALL = [["src/views","" ]]

# 项目配置 注意：使用 VITE_ 开头的变量会被 vite 自动加载
# ======================系统级别的配置==================================
# 项目title名称
VITE_TITLE = "ElcKer-Inter-SPA"
# =======================应用级别的配置=================================
# 系统的主页面
VITE_PAGE_HOME = '/ElckerApp'
# 系统的登录地址
VITE_PAGE_LOGIN = '/Login'
# 系统登录的请求地址
VITE_URL_LOGIN = '/elckerApi/authService/login'
# =======================应用安全级别的配置=================================
# 启用请求时的加签及相应时的验签功能 需要参见更新说明
VITE_APISATE= 'true'
# ==========================压缩配置======================================
# Whether to enable gzip or brotli compression
# Optional: gzip | brotli | none
# If you need multiple forms, you can use `,` to separate
VITE_BUILD_COMPRESS = 'none'

# Whether to delete origin files when using compress, default false
VITE_BUILD_COMPRESS_DELETE_ORIGIN_FILE = false

# 开发环境配置

# 日志的输出级别  'info' | 'warn' | 'error' | 'silent'
VITE_LOGLEVEL = "info"
# 是否使用 MOCK 数据
VITE_USE_MOCK = true
# 定义请求的基础URL, 方便跨域请求时使用 public path
VITE_BASE_API = ''
# 服务代理配置 此环境下无需配置但要存在
VITE_SERVER_PROXY = [["/dev-api","http://localhost:6001"]]
# 服务器地址
VITE_SERVER_HOST = "0.0.0.0"
# 服务器端口
VITE_SERVER_PORT = "6001"
# 是否使用https
VITE_SERVER_HTTPS = false
# cookie保存的域名，utils/cookie.js 要用  COOKIE 名称
VITE_COOKIE_NAME='localhost'

# 测试环境配置

# 日志的输出级别  'info' | 'warn' | 'error' | 'silent'
VITE_LOGLEVEL = "info"
# 是否使用 MOCK 数据
VITE_USE_MOCK = "false"
# 定义请求的基础URL, 方便跨域请求时使用 public path
VITE_BASE_API = '/dev-api'
# 服务代理配置 此环境下是必要的
VITE_SERVER_PROXY = [["/dev-api","https://localhost/"]]
# 服务器地址
VITE_SERVER_HOST = "0.0.0.0"
# 服务器端口
VITE_SERVER_PORT = "6001"
# 是否使用https
VITE_SERVER_HTTPS = false
# cookie保存的域名，utils/cookie.js 要用  COOKIE 名称
VITE_COOKIE_NAME='localhost'

# 生产环境配置


# 日志的输出级别  'info' | 'warn' | 'error' | 'silent'
VITE_LOGLEVEL = "info"
# 是否使用 MOCK 数据
VITE_USE_MOCK = true
# 定义请求的基础URL, 方便跨域请求时使用  public path 
# 使用生产mock 时为空字符串 不使用mock时，根据实际情况配置
VITE_BASE_API = ''
# 服务代理配置
VITE_SERVER_PROXY = [["/dev-api","https://localhost:6001"]]
# 服务器地址
VITE_SERVER_HOST = "0.0.0.0"
# 服务器端口
VITE_SERVER_PORT = "6001"
# 是否使用https
VITE_SERVER_HTTPS = false
# cookie保存的域名，utils/cookie.js 要用  COOKIE 名称 IP地址或者域名
VITE_COOKIE_NAME='localhost'
