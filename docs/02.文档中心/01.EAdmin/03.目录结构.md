## 目录及说明

``` sh
.
├── build # EAdmin 构建目录
│   ├── configs # 项目配置文件
│   ├── icon # 图标
│   ├── libs # 对三方的依赖 如lUI库
│   │   └── ElckerLibs # 三方依赖入口
│   ├── theme # 主题
│   ├── types # build 文件构建涉及的 类别存放目录
│   ├── utils # 工具包
│   ├── vite # VITE 封装构建目录
│   │   └── css # 样式
│   │   └── plugins # 插件
│   │   └── server # 代理服务配置
│   │   └── ViteConfigs # vite配置入口
├── public # 静态资源
├── mock # 模拟数据
│   └── _buildProdServer # 生产模拟数据入口
├── src # 项目代码
│   ├── api # 接口资源
│   ├── assets # 静态资源
│   ├── components # 组件
│   ├── router  # 路由配置
│   ├── store  # 数据存储
│   ├── views # 页面存放文件夹
│   │── │── About.vue # 简介页面
│   │── │── index.vue # 页面入口
│   │── │── error.vue # 错误页面
│   │── │── Login.vue # 登录页面
│   │── │── ElckerApp.vue # 业务文件布局页面
│   │── └── ElckerApp # 业务文件夹
│   │── │   └── index.vue # 业务入口
│   ├── types # 全局types
│   │── App.vue # 入口页面
│   └── main.js # 入口文件 加载组件 初始化等
├── .editorconfig #vscode编辑器配置，在启用prettier s时不起作用
├── env #全局环境配置
├── evn.development #开发环境配置
├── evn.production #正式环境配置
├── evn.test #测试环境配置
├── .eslintrc.js # eslint 配置项
├── .eslintignore # eslint 忽略配置项
├── .gitignore # git提交忽略配置项
├── .prettierignore # prettier 忽略配置项
├── .prettierrc # prettier 配置项
├── .stylelintignore # .stylelint 忽略配置项
├── stylelint.config # stylelint 配置项
├── commitlint.config # commitlint 配置项
├── tsconfig #ts配置
├── package.json # package.json
├── vite.config # vite 配置
```
